SET(OGRE_PATH "Ogre")
SET(OGRE_DOWNLOAD_DEST ${CMAKE_CURRENT_LIST_DIR}/Ogre_Rel_VS2015.7z)
IF(NOT EXISTS  ${OGRE_DOWNLOAD_DEST})
	file(DOWNLOAD 
	"https://drive.google.com/uc?authuser=0&id=1MFFmZaisy5OdsGTCFP5m0RAzhu-_nTrc&export=download" ${OGRE_DOWNLOAD_DEST})
ENDIF()
IF(NOT EXISTS ${CMAKE_CURRENT_LIST_DIR}/${OGRE_PATH})
    find_program(7Z_EXE 7z PATHS "$ENV{ProgramFiles}/7-Zip") 
	IF(7Z_EXE)
		set(ZIP_ARGS x -o${OGRE_PATH} -y Ogre_Rel_VS2015.7z)
		execute_process(COMMAND ${7Z_EXE} ${ZIP_ARGS} WORKING_DIRECTORY ${CMAKE_CURRENT_LIST_DIR})
	ENDIF(7Z_EXE)
ENDIF()
message(${CMAKE_CURRENT_LIST_DIR}/${OGRE_PATH})
SET(OGRE_SDK_PATH ${CMAKE_CURRENT_LIST_DIR}/${OGRE_PATH} CACHE FILEPATH "Path to ogre sdk root.")
